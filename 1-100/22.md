### 22. 括号生成

难度：Middle

相关话题：`字符串`、`回溯算法`

给出 *n* 代表生成括号的对数，请你写出一个函数，使其能够生成所有可能的并且 **有效的** 括号组合。



例如，给出 *n * = * * 3，生成结果为：





```
[
  "((()))",
  "(()())",
  "(())()",
  "()(())",
  "()()()"
]

```


```
/**
 * @param {number} n
 * @return {string[]}
 */
var generateParenthesis = function(n) {
    let s=''
    let l="(",r=")"
    let lN=0,rN=0
    let res=[]
    function dfs(s,n,lN,rN){
      if(lN&gt;n || lN&lt;rN)return
      if(lN===n &amp;&amp; rN===n)res.push(s)
      dfs(s+l,n,lN+1,rN)
      dfs(s+r,n,lN,rN+1)
    }
    dfs(s,n,lN,rN)
    return res
};



```
