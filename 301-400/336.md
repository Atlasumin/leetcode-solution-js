### 336. 回文对

难度：Hard

相关话题：`字典树`、`哈希表`、`字符串`

给定一组**唯一** 的单词， 找出所有***不同*** 的索引对 `(i, j)` ，使得列表中的两个单词， `words[i] + words[j]` ，可拼接成回文串。



**示例 1:** 



```
输入:

["abcd","dcba","lls","s","sssll"]
输出:

[[0,1],[1,0],[3,2],[2,4]] 
解释:

可拼接成的回文串为 ["dcbaabcd","abcddcba","slls","llssssll"]
```


**示例 2:** 



```
输入:

["bat","tab","cat"]
输出:

[[0,1],[1,0]] 
解释:

可拼接成的回文串为 ["battab","tabbat"]
```

```
/**
 * @param {string[]} words
 * @return {number[][]}
 */
var palindromePairs = function(words) {
  if(words.length<2)return []
  let hash={}
  for(let i=0;i<words.length;i++){
    hash[words[i]]=i
  }
  let res=[]
  for(let i=0;i<words.length;i++){
    for(let j=0;j<words[i].length;j++){
      let check1=words[i].substring(0,j),
          check2=words[i].substring(j,words[i].length)
      if(isPar(check1)){
        let search=check2.split('').reverse().join('')
        if(hash[search]!=null &amp;&amp; hash[search]!==i){
          if(search==='')res.push([i,hash[search]])
          res.push([hash[search],i])
        }
      }
      if(isPar(check2)){
        let search=check1.split('').reverse().join('')
        if(hash[search]!=null &amp;&amp; hash[search]!==i){
          if(search==='')res.push([hash[search],i])
          res.push([i,hash[search]])
        }        
      }
    }
  }
  return res
  function isPar(str){
    for(let i=0;i<str.length/2;i++){
      if(str[i]!==str[str.length-1-i])return false
    }
    return true
  }
}
```
