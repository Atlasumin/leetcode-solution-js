### 653. 两数之和 IV - 输入 BST

难度：Easy

相关话题：`树`

给定一个二叉搜索树和一个目标结果，如果 BST 中存在两个元素且它们的和等于给定的目标结果，则返回 true。



**案例 1:** 



```

输入:

 
    5
   / \
  3   6
 / \   \
2   4   7

Target = 9

输出:

 True
```






**案例 2:** 



```

输入:

 
    5
   / \
  3   6
 / \   \
2   4   7

Target = 28

输出:

 False
```





```
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} k
 * @return {boolean}
 */
var findTarget = function(root, k) {
    let res=false
    function preDSF(node){
        let cur=node
        if(search(k-cur.val,root,cur))return res=true
        if(!res &amp;&amp; node.left) preDSF(node.left)
        if(!res &amp;&amp; node.right) preDSF(node.right)
        return
    }
    function search(n,node,cur){
        if(node){
            if(n>node.val)return search(n,node.right,cur)
            else if(n<node.val)return search(n,node.left,cur)
            else if(node===cur) return search(n,node.right,cur)
            else return true
        }
        return false
    }
    preDSF(root)
    return res
};
```
